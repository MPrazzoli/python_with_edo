forcast2 = pd.DataFrame()

for i in forcast.index.get_level_values(0).unique():
    y = forcast.loc[[i]]
    y.reset_index(inplace=True)
    y.set_index(['Snapshot'], inplace=True)
    y.index = y.index.to_timestamp('M')
    y.sort_index().asfreq(freq='M')
    idx = pd.date_range(min(y.index), max(y.index), freq='M')
    y = y.reindex(idx, fill_value=None)
    y.index = y.index.to_period('M')
    y_fill = y.bfill()
    
    forcast2 = forcast2.append(y_fill)

def difference(dataset):
	diff = list()
	for i in range(1, len(dataset)):
		value = dataset[i] - dataset[i - 1]
		diff.append(value)
	return diff
    #return pd.Series(diff)

Autodeterminazione parametri del modello

All'interno del ciclo for, viene applicata la grid search per trovare l'insieme ottimale di parametri che produce le migliori prestazioni per il nostro modello ARIMA. Dato un insieme di modelli candidati per i dati, il modello preferito quello con il valore minimo di Akaike information criterion (AIC).

Fit del modello e forcast di 2 mesi successivi all'ultimo disponibile

Una volta determinati i parametri ARIMA migliori per la serie storica, applico il modello alla serie appendo al dataframe di partenza (X) il valore predetto dal modello (y) nei due mesi successivi.

NB:se la serie storica di partenza non stazionaria il parametrodsar maggiore 0, pertanto la serie storica verr differenziata. In questo caso necessario riportare le predizioni alla scala originale prima di poterle appendere alla serie

import warnings
warnings.filterwarnings('ignore')

p=d=q=range(0,5)
pdq = list(itertools.product(p,d,q))

for i in forcast2.index.get_level_values(0).unique():
    X = forcast2.loc[[i]]
    X.reset_index(level=[1], inplace=True)
    X.set_index(['snapshot'], inplace=True)
    X = X.asfreq(freq='M')
    a = 0
    
    for param in pdq:
        try:
            model_arima = ARIMA(X, order=param)
            model_arima_fit = model_arima.fit()
            if model_arima_fit.aic < a:
                a = model_arima_fit.aic
                best_param = param
        except:
            continue
            
    print('Dealer Code: ' + i)
    print('parametri modello ARIMA: ', best_param)
    
    try:
        model = ARIMA(X, order=(best_param))
        result = model.fit(disp=False)
        result.plot_predict(start=X.index.min()+1, end=X.index.max()+5, dynamic=False)
        plt.ylabel('Score')
        plt.ylabel('Snapshot')
        plt.title("Dealer's time series with forecast")
        plt.show()
        print(result.summary())
   
        y = result.predict(start=X.index.max()+1, end=X.index.max()+5)
        y = pd.DataFrame(y)
        y.columns = ['score']

        if (best_param[1]>0):
            y = y.cumsum()
            y = y.add(X.iloc[-1])
        
        final_score = X.append(pd.DataFrame(y))
        
        y.rename(columns={'score': 'forecast'}, inplace=True)
        print(X)
        print(y)
        print('___________________________________________________________________________________________')
    except:
        print('modello non applicabile alla serie storica')
        print('___________________________________________________________________________________________')
        continue
